switch문!!
{: .notice--primary}


```c
#include <stdio.h>
#include <string.h>

int main() {
    char a[5] = "a1a20"; // 문자열 ""로 적으면 마지막에 자동으로 널이 붙음 
    printf("a의 길이: %ld\n", strlen(a));
    char b[6] = {0, };
    int i, j=0;
    
    for(i=0; i<5; i++){
        switch(a[i]){
            case 48: 
            case 49: 
            case 50:
                b[j++] = a[i];
                break;
        }
    }
    
    for(i=0; b[i]!=0; i++)
        printf("%d\t", b[i] - 48);

    return 0;
}
```
